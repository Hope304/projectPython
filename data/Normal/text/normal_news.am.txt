Առաջիկայում սպասվում են քաղաքական զարգացումներ, որոնց պետք է միասին դիմագրավենք․ Իշխան Սաղաթելյան Առանց չափազանցնելու, դրված է պետականության գոյության հարցը... Թուրքիան պետք է հայերից ներողություն խնդրեր 1915–ի համար. Թուրքիայի նախկին փոխվարչապետ Իմ պաշտոնավարման օրոք Թուրքիան առաջին երկրներից է, որ ճանաչեց Հայաստանի անկախությունը... Ադրբեջանում Լեռնային Ղարաբաղի խորհրդարանի քանդված շենքն «ապօրինի» է ճանաչվել Որոշում է կայացվել քանդել անկախության տարիներին կառուցված բոլոր շենքերը... Հատուկ նախագծեր Փակել Ամբերդ՝ Արագածի սեւ մարգարիտը (ֆոտո) Արագած լեռան լանջում թաքնված է Հայաստանի ճարտարապետական ​​պատմության գանձերից մեկը՝ Ամբերդի ամրոցը... Գրքերի ուշագրավ աշխարհ. Հայաստանի խոշորագույն գիտական գրադարանն 85 տարեկան է (ֆոտո) Մեծամոր. հնագույն իշխանության ծաղկող մայրաքաղաքը (ֆոտո) Անցած դարերի ժառանգություն. թուրքմեն էմիրի դամբարան Երեւանի արվարձանում (ֆոտո) Երկաթուղային տեխնիկայի թանգարան՝ Գյումրիում․ Հնի եւ նորի միավորում (ֆոտո) Հայկական դիվիզիաները Մեծ հայրենական պատերազմում՝ Կովկաս, Ստալինգրադ, հաղթական քոչարի Ռեյխսթագի պատերի տակ (Ֆոտո) «Ջոջերի բոցերը». NEWS.am-ի ֆոտոնախագծում Մանվել Գրիգորյանն է «Ջոջերի բոցերը». NEWS.am-ի ֆոտոնախագծում Սամվել Ալեքսանյանն է 119 տարի առաջ Հայաստան ժամանեց առաջին գնացքը. ռեպորտաժ՝ Երկաթգծի թանգարանից (ֆոտո) Ռազմապատմական թանգարան. Արցախյան պատերազմի եւ մեր ժամանակի հերոսները (ֆոտոռեպորտաժ) Հին լուսանկարներ, համազգեստներ, զենք. Հայաստանի Ոստիկանության թանգարանը այցելուներին է սպասում (ֆոտո) «Ջոջերի բոցերը». NEWS.am-ի ֆոտոնախագծում Առաքել Մովսիսյանն է «Ջոջերի բոցերը». NEWS.am-ի նոր ֆոտոնախագիծը Անավարտ պատերազմում ընկածների հիշատակին (մաս 6) Անավարտ պատերազմում ընկածների հիշատակին (մաս 5) Բոլորը Հայաստանը պահում է պատժամիջոցային ռեժիմներն առանց դրանց միանալու. Կենտրոնական բանկ Հայաստանը որեւէ պատժամիջոցային ծրագրի չի միացել... Հայաստան եկած ՏՏ ընկերություններն արդեն գնում են Վրաստան․ Սևակ Հովհաննիսյան Ո՞վ է խոսում միջանցք տալու մասին, ամբողջ պայքարը գնում է այդ միջանցքի համար, որ չպիտի տանք․ Սասուն Միքայելյան Բեզոսը 2021-ից հետո առաջին անգամ գլխավորել է մոլորակի ամենահարուստ մարդկանց ռեյթինգը. Bloomberg Ադրբեջանը 34 մլն դոլարի օգնություն է ուղարկել Ուկրաինային ՀՀ իշխանությունները չեն հասկանում՝ ինչով կավարտվի, եթե ռուսները վերջնականապես հանեն անվտանգության բարձիկները. Կիվիրյան Բոլորս տեսանք, թե հայ խոպանչիներին որոշումը ինչպիսի բունտեր բարձրացրեց... Բարերար Վահե Բադալյանը 15-րդ դարի մեծարժեք ձեռագիր է նվիրաբերել Մատենադարանին (վիդեո, ֆոտո) «Անտեսված ենք ու չենք կարողանում ապրել». Քարվաճառի ու Քաշաթաղի բնակիչների բողոքի ակցիան Ադրբեջանցի պատգամավորը առաջարկում է գործողություն իրականացնել Հայաստանի Տավուշի մարզում Ադրբեջանը որ Ղարաբաղը հայաթափեց, էդ քարտեզները մնաց Ստեփանակերտում, թող գնան, հավաքեն․ Գագիկ Մելքոնյան ԱՄՆ պետդեպարտամենտը մեկնաբանել է Սերգեյ Լավրովի հայտարարությունը Հայաստանի մասին Claude 3. Anthropic-ն ստեղծել է ԱԲ նոր մոդելներ, որոնք գերազանցում են Google-ի և OpenAI-ի մոդելներին Ռուսները բաց տեքստով մեզ ասում են՝ կողմնորոշվեք. Ընդ որում, այդ կոչը ուղղված է նաեւ ժողովրդին. Մեզ չեն խղճալու. Կիվիրյան բեռնել ավելին Օրվա տեսանյութը Ի՞նչ խնդիրներ է առաջացնում էլեկտրոնային հայցադիմումների ծրագիրը․ հարթակից դժգոհություններ կան Այլ տեսանյութեր Ամբողջը 15:28 Հայաստան եկած 90 հազար «այթիշնիկներից» 15-20 հազարն է մնացել Հայաստանում. Հովիկ Մուսայելյան 10:49 Մենք ՀԱՊԿ–ից դեռեւս դուրս չենք եկել, թե ինչի կհանգեցնի դուրս գալը, գաղտնիք է պարունակում. ԳՇ պետ 15:41 Նիկոլ Փաշինյանը ժամանեց Եգիպտոս Տեսանյութեր Ամբողջը Մի քանի անգամ Իրանը զորքերը մոտեցրել է Սյունիքի սահման ու կոշտ զգուշացրել է. Կիվիրյան Նիկոլ Փաշինյանը պետք է անհապաղ ձերբակալվի. Պարույր Հայրիկյան Հայաստանը մտադիր է հումանիտար օգնություն հասցնել Գազա. Փաշինյան Աճպարարություն կարմիր գծի վրա՞, թե՞ արտաքին վեկտորի կտրուկ անցում դեպի ԵՄ Մենք չենք հանձնվելու, միասին փորձելու ենք գնալ դեպի հաղթանակ. Լեւոն Քոչարյան «Եռաբլուրում գոնե մի տեղ տան, մեր տղաների նկարները դնենք, գնանք ծաղիկ դնենք». զոհված արցախցու մայր Քաղաքականություն Ամբողջը Ավտովթարի արդյունքում մահացած զինծառայողի հայրը այդպես էլ ընդունելության չարժանացավ Կարեն Անդրեասյանի կողմից. փաստաբան ՀՀ քաղաքացու դիմումով արտահայտված խնդրանքը շատ պարզ էր՝ որդու մահվան վերաբերյալ քրեական գործը քննող… Վազգենն էլ մեղք ունի, երբ որ զինգրքույկ էր ուղարկում այս գրչակին, լրագրողին՝ բանակից ազատվելու համար. Աղվան Վարդանյան Մենք մեր աջակցությունն ենք հայտնում հայ-ադրբեջանական երկխոսությանը և բանակցային պրոցեսին. Եգիպտոսի նախագահ ԵՄ-ն կարևորում է Արցախի հայերի վերադարձի համար պայմանների ստեղծումն ու իրենց տներում արժանապատիվ ապրելու իրավունքը 2020-ից հետո դժվար է հասկանալ մարդկանց, որոնք էս իշխանությանը հավատում են․ Լևոն Քոչարյան Արցախի ԱԺ-ն դատապարտում է Ադրբեջանի վանդալիստական քաղաքականությունը. հայտարարություն Տնտեսություն Ամբողջը Հայկական ինքնաթիռների՝ դեպի Եվրոպա թռիչքների արգելքը վերացնելու գործընթացը մոտ է Անդրադառնալով դեպի Եվրամիության երկրներ թռիչքներ իրականացնելու արգելքի վերացման ժամկետներին՝ Փայասլյանը նշեց, որ հստակ ժամկետներ նշելը դեռևս... Հայաստանը միջազգային ներդրողների համար գրավիչ լինելու լավ շանս ունի. «Freedom Holding Corp.» ՊԵԿ պատվիրակությունը մասնակցել է ԵԱՏՄ պետությունների մաքսային ծառայությունների միավորված կոլեգիայի 45-րդ նիստին Բրեթ Քինգը Doing Digital միջազգային ֆորումի գլխավոր խոսնակն է (ֆոտո) Կառավարության պարտքի ծավալի բաժինը դրամով ավելացել է՝ 20 տոկոսից հասնելով 46 տոկոսի. ֆիննախ Գևորգ Պապոյանը նշանակվել է էկոնոմիկայի նախարար Վերլուծություն Ամբողջը Վերևից թելադրված որոշման տակ մարդկային կյանքեր են. Էլինար Վարդանյան Կալանքի միջոցով փորձում են մարդուն պատժել, ճնշել, կոտրել, ավելի կոմպրոմիսային սարքել, ամենևին կարևոր չէ՝ կալանքի հիմք կա, թե ոչ... Անհատ ռիելթորները կբերվեն հարկային դաշտ. նոր օրենսդրական նախաձեռնություն Հնդկաստանը մտահոգված է Ադրբեջանի, Պակիստանի և Թուրքիայի միջև աճող մերձեցմամբ. Bharat Shakti Ապրիլին Ռուսաստանը կարող է Հայաստանի նկատմամբ հստակ գործողությունների անցնել. փորձագետ Մակրոնի և Շոլցի «ջրերը նույն առվով չեն հոսում». Վերջինիս 20-ամյա վարորդը հոսպիտալացվել է Մշակույթ Ամբողջը Ստեփանակերտի երաժշտական դպրոցի սաների համերգը՝ Կոմիտասի անվան կամերային երաժշտության տանը 1938 թ․ Ստեփանակերտում հիմնադրված երաժշտական դպրոցում ի սկզբանե գործում էին 3՝ դաշնամուրային, լարային և թառի բաժինները, սովորում էր ընդամենը... Մատենադարանին է նվիրաբերվել Հովհաննես երեցի ընդօրինակած Ավետարանը «Տերյան» մշակութային կենտրոնի «My way» նախագիծը ներառվել է ՅՈՒՆԵՍԿՕ-ի IFCD դրամաշնորհային ծրագրում Միջազգային փառատոներում մրցանակի արժանացած գործիչները դրամական խրախուսում կստանան. Ամսական 130-800 հազար դրամ Առաջին անգամ Անկախ Հայաստանի պատմության ընթացքում թանգարանների ոլորտն ունի իր մայր օրենքը 88-ամյա Ալեն Դելոնից 72 միավոր հրազեն է առգրավվել Ավտոաշխարհ Ամբողջը Գիտական ​​հայտնագործությունը կարող է ստիպել էլեկտրական մեքենաներին մեկ լիցքավորմամբ 1000 կմ անցնել Նոր հետազոտության հեղինակները որոշել են աշխատել սիլիցիումի մասնիկների հետ, որոնք մոտ 1000 անգամ ավելի մեծ են, քան նանոմասնիկները... Հայտնվել է թարմացված Lamborghini Urus-ի առաջին լուսանկարը Rolls-Royce-ը ներկայացրել է աշխարհի ամենաթանկ ավտոմեքենան Վաճառվել է 7 «գոլդ» համարանիշ, ընդհանուր արժեքը՝ 156 միլիոն 200 հազար դրամ Չինաստանում Միկի Մաուսի ականջներով ավտոմեքենա է արտադրվել 2-րդ գիծ կանգնելու համար տուգանային միավորը 0.5-ից կդառնա 0.25, «պլյոնկայի» համար էլ 2-րդ անգամ կհանվի 2 միավոր Sport Ամբողջը ՉԼ․ ՊՍԺ-ն ու «Բավարիան» դուրս եկան քառորդ եզրափակիչ (տեսանյութ) «Ինտերը» հաղթեց եւ արդեն 15 միավոր առաջ է «Յուվենտուսից» (տեսանյութ) «Արսենալը» խնդիրներ չունեցավ՝ 6:0 (տեսանյութ) «Փյունիկի» կամային հաղթանակը (տեսանյութ) Style Ամբողջը Սյուզաննա Մելքոնյանը կներկայացնի Հայաստանը «Славянский базар 2024» մցույթում Արուս Տիգրանյանն ամուսնու հետ տեսանյութ է հրապարակել ու խոսել միասին անցկացրած «ամենադժվար 1 տարվա մասին» «Բարի օր»՝ տաքսիստ Հենդոյի հետ. Մհեր Մխիթարյանի հայտնի կերպարի մասին ֆիլմի պրեմիերան ու կարմիր գորգը (ֆոտո/վիդեո) «Ինձ դուր չէր գալիս իմ հաջողությունը, ես խուսափում էի դառնալ մոդայիկ նկարիչ». Մարտիրոս Սարյանի ծննդյան օրվա առիթով Medicine Ամբողջը X հիվանդության համաճարակ. ե՞րբ է այն սպասվում, ի՞նչ հարուցիչից է այն առաջանալու. զրույց Մերի Տեր-Ստեփանյանի հետ Հետազոտությունը պարզել է ավելի քան 275 մլն նոր գենետիկ տեսակներ Քովիդը մարդու ուղեղից մինչև 20 տարվա կյանք է խլում Հայաստանում արյունաբանական չարորակ հիվանդության դեպքում առաջին անգամ կատարվել է ոսկրածուծի ալոգեն փոխպատվաստում Tech Ամբողջը Այլ երկրներն ավելի բարենպաստ պայմաններ են առաջարկում. ՏՏ ընկերությունները ինչո՞ւ կարող են հեռանալ ՀՀ-ից և ի՞նչ կարելի է անել դրա դեմ ROG Zephyrus G16․ նորույթ փորձառու գեյմերների համար Կրիպտոյի տակ հավատն է, ինչպես փողի կամ ոսկու․ Վիգեն Առուշանյան Ավելի քան 500 միլիոն մարդ շատ շուտով ռոմանտիկ հարաբերություններ կունենա ոչ կենսաբանական ուղեկիցների հետ. Դավիթ Յան Մամուլի տեսություն Ամբողջը «Ժողովուրդ» Ալեն Սիմոնյանի քավորին իշխանությունն օգնում է, բնությունը՝ խանգարում. Ծաղկաձորի «Պալլադան» խնդիրներ ունի «Հրապարակ» Ընդդիմադիր եւ իշխանական կանանց երկխոսությունը չի ստացվել «Ժողովուրդ» «Քիմիկների» գործով դատարան կգնա գլխավոր շտաբի պետը «Հրապարակ» «Ասել եմ՝ մարտի 1-ից, ուրեմն մարտի 1-ից» «Փաստ» «Անճար-միամիտ» ձևացողների փորձանքաբեր «փորձարարությունները» «Հրապարակ» Արցախցիների աջակցության գումարները կկրճատվեն «Փաստ» Փիառային «դեսանտ» Լոս Անջելեսում «Հրապարակ» Կառավարությունը մի ողջ թաղամաս հանրային գերակա շահ է ճանաչել «Ժողովուրդ» Պայմանագիրը խզվել է, շինարարությունը դադարել․ շինարարական ոլորտում իշխանության հերթական ձախողումը Լոռիում է «Հրապարակ» ՊՆ-ն
